## 1. 快速排序 ⭐⭐⭐

```cpp
// 递归版（面试最常写）
void quickSort(vector<int>& nums, int l, int r) {
    if (l >= r) return;
    int i = l - 1, j = r + 1, pivot = nums[(l + r) >> 1];
    while (i < j) {
        do i++; while (nums[i] < pivot);
        do j--; while (nums[j] > pivot);
        if (i < j) swap(nums[i], nums[j]);
    }
    quickSort(nums, l, j);
    quickSort(nums, j + 1, r);
}
```

> 考点：partition 写法、`(l + r) >> 1` 防溢出、递归边界

## 2. 归并排序 ⭐⭐

```cpp
void mergeSort(vector<int>& nums, int l, int r, vector<int>& tmp) {
    if (l >= r) return;
    int mid = (l + r) >> 1;
    mergeSort(nums, l, mid, tmp);
    mergeSort(nums, mid + 1, r, tmp);
    int i = l, j = mid + 1, k = l;
    while (i <= mid && j <= r) {
        if (nums[i] <= nums[j]) tmp[k++] = nums[i++];
        else tmp[k++] = nums[j++];
    }
    while (i <= mid) tmp[k++] = nums[i++];
    while (j <= r) tmp[k++] = nums[j++];
    for (i = l; i <= r; i++) nums[i] = tmp[i];
}
```

> 考点：分治合并、辅助数组、应用于链表排序和逆序对

## 3. 反转链表（迭代 + 递归）⭐⭐⭐

```cpp
// 迭代版
ListNode* reverseList(ListNode* head) {
    ListNode *pre = nullptr, *cur = head;
    while (cur) {
        ListNode* nxt = cur->next;
        cur->next = pre;
        pre = cur;
        cur = nxt;
    }
    return pre;
}

// 递归版
ListNode* reverseListRecursive(ListNode* head) {
    if (!head || !head->next) return head;
    ListNode* newHead = reverseListRecursive(head->next);
    head->next->next = head;
    head->next = nullptr;
    return newHead;
}
```

## 4. K 个一组翻转链表 ⭐⭐⭐

```cpp
ListNode* reverseKGroup(ListNode* head, int k) {
    ListNode* dummy = new ListNode(0, head);
    ListNode *pre = dummy, *end = dummy;
    while (end->next) {
        for (int i = 0; i < k && end; i++) end = end->next;
        if (!end) break;
        ListNode* start = pre->next;
        ListNode* nxt = end->next;
        end->next = nullptr;
        pre->next = reverseList(start);
        start->next = nxt;
        pre = start;
        end = pre;
    }
    return dummy->next;
}
```

> 考点：组内反转 + 重新连接，细节多容易出 bug

## 5. LRU Cache ⭐⭐⭐

```cpp
class LRUCache {
    int cap;
    list<pair<int, int>> cache;          // key, value
    unordered_map<int, list<pair<int,int>>::iterator> mp;

public:
    LRUCache(int capacity) : cap(capacity) {}

    int get(int key) {
        auto it = mp.find(key);
        if (it == mp.end()) return -1;
        cache.splice(cache.begin(), cache, it->second); // 移到头部
        return it->second->second;
    }

    void put(int key, int value) {
        auto it = mp.find(key);
        if (it != mp.end()) {
            it->second->second = value;
            cache.splice(cache.begin(), cache, it->second);
            return;
        }
        if (cache.size() == cap) {
            mp.erase(cache.back().first);
            cache.pop_back();
        }
        cache.emplace_front(key, value);
        mp[key] = cache.begin();
    }
};
```

> 考点：双向链表 + 哈希表、`splice` 用法、头部最新尾部最旧

## 6. 手写线程安全单例（C++11）⭐⭐

```cpp
class Singleton {
public:
    static Singleton& getInstance() {
        static Singleton instance;  // C++11 保证线程安全
        return instance;
    }
private:
    Singleton() = default;
    Singleton(const Singleton&) = delete;
    Singleton& operator=(const Singleton&) = delete;
};
```

> 考点：`static` 局部变量线程安全（C++11 magic static）、删除拷贝/赋值

## 7. 二叉树非递归遍历（三种）⭐⭐⭐

```cpp
// 前序：根左右（栈：先压右，再压左）
vector<int> preorder(TreeNode* root) {
    vector<int> res;
    if (!root) return res;
    stack<TreeNode*> stk; stk.push(root);
    while (!stk.empty()) {
        TreeNode* node = stk.top(); stk.pop();
        res.push_back(node->val);
        if (node->right) stk.push(node->right);
        if (node->left)  stk.push(node->left);
    }
    return res;
}

// 中序：左根右（一路向左压栈）
vector<int> inorder(TreeNode* root) {
    vector<int> res;
    stack<TreeNode*> stk;
    TreeNode* cur = root;
    while (cur || !stk.empty()) {
        while (cur) { stk.push(cur); cur = cur->left; }
        cur = stk.top(); stk.pop();
        res.push_back(cur->val);
        cur = cur->right;
    }
    return res;
}

// 后序：左右根 → 反向就是 根右左（改前序：先左后右）
vector<int> postorder(TreeNode* root) {
    vector<int> res;
    if (!root) return res;
    stack<TreeNode*> stk; stk.push(root);
    while (!stk.empty()) {
        TreeNode* node = stk.top(); stk.pop();
        res.push_back(node->val);
        if (node->left)  stk.push(node->left);   // 先左
        if (node->right) stk.push(node->right);  // 后右
    }
    reverse(res.begin(), res.end());
    return res;
}
```

## 8. 生产者-消费者（C++ 线程）⭐⭐

```cpp
// 用 mutex + condition_variable
mutex mtx;
condition_variable cv;
queue<int> q;
const int MAX_SIZE = 10;

void producer(int id) {
    for (int i = 0; i < 100; i++) {
        unique_lock<mutex> lock(mtx);
        cv.wait(lock, [] { return q.size() < MAX_SIZE; });
        q.push(i);
        cout << "P" << id << " → " << i << endl;
        lock.unlock();
        cv.notify_all();
    }
}

void consumer(int id) {
    while (true) {
        unique_lock<mutex> lock(mtx);
        cv.wait(lock, [] { return !q.empty(); });
        int val = q.front(); q.pop();
        cout << "C" << id << " ← " << val << endl;
        lock.unlock();
        cv.notify_all();
    }
}
```

## 9. 手写智能指针（shared_ptr 简化版）⭐⭐

```cpp
template<typename T>
class SimpleSharedPtr {
    T* ptr;
    int* refCount;
public:
    explicit SimpleSharedPtr(T* p = nullptr)
        : ptr(p), refCount(new int(1)) {}

    SimpleSharedPtr(const SimpleSharedPtr& other)
        : ptr(other.ptr), refCount(other.refCount) {
        (*refCount)++;
    }

    SimpleSharedPtr& operator=(const SimpleSharedPtr& other) {
        if (this != &other) {
            release();
            ptr = other.ptr;
            refCount = other.refCount;
            (*refCount)++;
        }
        return *this;
    }

    ~SimpleSharedPtr() { release(); }

    T* operator->() { return ptr; }
    T& operator*()  { return *ptr; }
    int use_count() { return *refCount; }

private:
    void release() {
        if (--(*refCount) == 0) {
            delete ptr;
            delete refCount;
        }
    }
};
```

## 10. 字符串转整数（atoi）⭐

```cpp
int myAtoi(string s) {
    int i = 0, n = s.size(), sign = 1;
    long long res = 0;
    while (i < n && s[i] == ' ') i++;           // 去空格
    if (i < n && (s[i] == '+' || s[i] == '-'))  // 判符号
        sign = (s[i++] == '-') ? -1 : 1;
    while (i < n && isdigit(s[i])) {
        res = res * 10 + (s[i++] - '0');
        if (res * sign > INT_MAX) return INT_MAX;   // 溢出处理
        if (res * sign < INT_MIN) return INT_MIN;
    }
    return (int)(res * sign);
}
```
